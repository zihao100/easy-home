// store.js
import Vue from 'vue'
import Vuex from 'vuex'
const state = {
  num:1
}
const getters = {
    count(state, getters){
        return num+=1;
    }
}
const mutations = {
// mutations不要有异步，繁琐操作和异步在actions中执行，mutations修改数据
    setData(state, param){
        state.num = param;
    }
}
const actions = {
    // 组件调用，接收一个参数param
    getData(store,  param){
     // 假设数据两秒后才请求到
       setTimeout(() => {
         let newNum = 112;
         // 提交事件，传参只支持一个参数，可以写成对象
         store.commit('setData', newNum);
      }, 2000);
   }
}
export default new Vuex.Store({
  state,
  getter,
  mutations,
  actions
});
```

// 组件
<template>
    <div>
        {{$store.state.num}}// 省略this
        {{count}}
    </div>
</template>
<script>
// 简写方式需要引入：
import {mapState,mapGetters,mapMutations,mapActions} from 'vuex'
export default {
     data(){
        return {
        }
    },
    computed: {
        // 获取state.num
        num(){
            return this.$store.state.num;
        },
        // 获取getters.count
        count(){
            this.$store.getters.count
        },
        // 简写
        ...mapState(['num']),// 对象方式{num: (state)=>state.num}
        ...mapGetters(['count']),
        ...mapMutations(['setData'])
    },
    methods: {
        setData(param){
            // 调用mutations,第一个参数事件名，第二个参数传入的参数
            // this.setData(param)
            this.$store.commit('setData', param);
        },
        getData(param){
            // 调用actions,第一个参数事件名，第二个参数传入的参数
            this.$store.dispatch('getData', param);
        },
        // 简写
        ...mapMutations(['setData']),// this.$store.commit('setData')
        ...mapActions(['getData'])
    }
}
</script>
```

// 简写
import {mapState,mapGetters,mapMutations,mapActions} from 'vuex'
computed: {
        // 如果访问模块下的state，只能用下面的方式访问
        // 全局 name 转为组件 name 通过组件中通过this.name访问
        ...mapState({
            name: (state)=>state.m.name
        }),
        ...mapGetters({
            num: 'm/count'
        })
    }
methods: {
        // 将全局的mmapActions事件，转为组件的methods事件
        ...mapActions({
            setTitle: 'm/requestTitle'
        }),
        // 将全局的mutations事件，转为组件的methods事件
        ...mapMutations({
            modifyTitle: 'm/changeTitle'
        })
    }

// 模块化
const store = new Vuex.Store({
  modules: {
      // 模块名account
    account: {
      // 通过添加 namespaced: true 的方式使其成为带命名空间的模块  
      namespaced: true,
      // 模块内容（module assets）
      state: { ... }, // 模块内的状态已经是嵌套的了，使用 `namespaced` 属性不会对其产生影响
      getters: {
        isAdmin () { ... } // -> getters['account/isAdmin']
      },
      actions: {
        login () { ... } // -> dispatch('account/login')
      },
      mutations: {
        login () { ... } // -> commit('account/login')
      },

      // 嵌套模块
      modules: {
        // 继承父模块的命名空间
        myPage: {
          state: { ... },
          getters: {
            profile () { ... } // -> getters['account/profile']
          }
        },

        // 进一步嵌套命名空间
        posts: {
          namespaced: true,
          state: { ... },
          getters: {
            popular () { ... } // -> getters['account/posts/popular']
          }
        }
      }
    }
  }
})
